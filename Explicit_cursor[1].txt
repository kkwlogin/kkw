DELIMITER //
CREATE PROCEDURE CURSR()
BEGIN
    DECLARE done INT DEFAULT FALSE;
    DECLARE ID INT;
    DECLARE NAM VARCHAR(200);
    DECLARE EMP_CURSR CURSOR FOR
        SELECT EMPLOYEE_ID , NAME FROM EMPLOYEES;
    DECLARE CONTINUE HANDLER FOR NOT FOUND SET done = TRUE;
    OPEN EMP_CURSR;
    READ_LOOP: LOOP
        FETCH EMP_CURSR INTO ID , NAM ;
        IF done THEN
            LEAVE READ_LOOP;
        END IF;
        SELECT CONCAT('EMPLOYEE ID ', ID , ',NAME' , NAM);
    END LOOP;
    CLOSE EMP_CURSR;
END;
//

DELIMITER ;
CALL CURSR();
